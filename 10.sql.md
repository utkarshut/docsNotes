//find employee whose salary is greater than their manager's salary

SELECT e.name
FROM Employee e
JOIN Employee m
ON e.managerId = m.id
WHERE e.salary > m.salary;

- Question: Find all customers who placed more than 3 orders in the last 30 days.
    Table1: Customers (customer_id, name, city, created_date)


    Table2: Orders (order_id, customer_id, order_date, amount, status)
            SELECT 
            c.customer_id, 
            c.name, 
            COUNT(o.order_id) AS total_orders
        FROM Customers c
        JOIN Orders o ON c.customer_id = o.customer_id
        WHERE o.order_date >= CURRENT_DATE - INTERVAL '30' DAY
        GROUP BY c.customer_id, c.name
        HAVING COUNT(o.order_id) > 3;
